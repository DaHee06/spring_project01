CREATE TABLE BOARD (
  BOARD_NO INT NOT NULL AUTO_INCREMENT COMMENT 'BOARD_NO',
  BOARD_TITLE VARCHAR(50) COMMENT '제목',
  WRITER_ID VARCHAR(20) COMMENT '작성자 아이디',
  WRITER_NICK VARCHAR(20) COMMENT '작성자 닉네임',
  BOARD_PWD VARCHAR(20) COMMENT '비밀번호',
  BOARD_PWD2 VARCHAR(20) COMMENT '비밀번호 확인',
  BOARD_CONTENT VARCHAR(4000) COMMENT '내용',
  BAORD_HIT INT COMMENT '조회수',
  REG_DT DATETIME COMMENT '생성일시',
  UPD_DT DATETIME COMMENT '수정일시',
  C_NO INT COMMENT '카테고리 번호',
  PRIMARY KEY(BOARD_NO)
);


CREATE TABLE CATEGORY (
  C_NO INT NOT NULL AUTO_INCREMENT COMMENT 'C_NO',
  C_NAME VARCHAR(20) COMMENT '카테고리 이름',
  PRIMARY KEY(C_NO)
);


CREATE TABLE FILE (
  FILE_NO INT NOT NULL AUTO_INCREMENT COMMENT 'FILE_NO',
  ORG_FILE_NM VARCHAR(50) COMMENT '첨부파일명',
  SAVE_FILE_NM VARCHAR(50) COMMENT '저장명',
  FILE_PATH VARCHAR(50) COMMENT '경로',
  REG_DT DATETIME COMMENT '생성일시',
  UPD_DT DATETIME COMMENT '수정일시',
  BOARD_NO INT COMMENT 'BOARD_NO',
  PRIMARY KEY(FILE_NO)
);


CREATE TABLE COMMENT (
  CMT_NO INT NOT NULL AUTO_INCREMENT COMMENT 'CMT_NO',
  CMT_ID VARCHAR(20) COMMENT '댓글 작성자',
  CMT_CONTENT VARCHAR(100) COMMENT '댓글 내용',
  REG_DT DATETIME COMMENT '생성일시',
  UPD_DT DATETIME COMMENT '수정일시',
  BOARD_NO INT COMMENT 'BOARD_NO',
  PRIMARY KEY(CMT_NO)
);
